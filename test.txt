        const applianceLabels = [

            'Dishwasher',
            'Air Conditioner',
            'Oven',
            'Television',
            'Microwave',
            'Lighting',
            'Computer',
            'Heater'
        ];

        const energyData = [
            400, // Refrigerator

            150, // Dishwasher
            600, // Air Conditioner
            200, // Oven
            180, // Television
            100, // Microwave
            300, // Lighting
            220, // Computer
            350  // Heater
        ];

        // Configuration for the bar chart
        const data = {
            labels: applianceLabels,
            datasets: [{
                label: 'Energy Consumption (kWh/year)',
                data: energyData,
                backgroundColor: [
                    'rgba(54, 162, 235, 0.7)',   // Refrigerator
                    'rgba(255, 99, 132, 0.7)',   // Washing Machine
                    'rgba(255, 206, 86, 0.7)',   // Dishwasher

                    'rgba(153, 102, 255, 0.7)',  // Oven

                    'rgba(199, 199, 199, 0.7)',  // Microwave
                    'rgba(83, 102, 255, 0.7)',   // Lighting
                    'rgba(255, 102, 255, 0.7)',  // Computer
                    'rgba(102, 255, 102, 0.7)'   // Heater

                    'rgba(255, 206, 86,1)',

                    'rgba(153, 102, 255,1)',

                    'rgba(83, 102, 255,1)',
                    'rgba(255, 102, 255,1)',
                    'rgba(102, 255, 102,1)'
                ],
                borderWidth: 1
            }]
        };

            type: 'bar',
            data: data,

                            size: 18
                        }
                    },

                                if (label) {
                                    label += ': ';
                                }
                                label += context.parsed.y + ' kWh/year';
                                return label;
                            }
                        }

                },
                scales: {
                    x: {

                            display: true,

                                size: 12

                            }

                }

        const energyChart = new Chart(ctx, config);
